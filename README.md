# Install
```js
npm install twlistview --save
```
```js
npm install lottie-react-native --save
```

# Getting started

```js
import TWListView from "twlistview";

```

```js
<TWListView
        isLoading={this.state.isLoading}
        ref={(ref) =>{
          this.TWListView = ref;
        }}
        onLoaded={() =>{

            this.load();
          }}
        renderRow={rowData => {
            return (<View/>)
        }
 />
 ```
 
 # Stop loading
 ```js
 this.TWListView.stopLoading();
  ```
